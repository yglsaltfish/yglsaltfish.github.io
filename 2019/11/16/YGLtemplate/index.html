struct Tree

    int l, r;
    int sum;
    int getmid() { return (l + r) / 2; }
} node[maxn << 2];
int a[maxn];
void build(int rt,int l,int r)
{
    node[rt].l = l, node[rt].r = r;
    if(l == r)
    {
        node[l].sum = a[l];
        return;
    }
    int mid = (l + r) / 2;
    build(rt << 1, l, mid);
    build((rt << 1) | 1, mid + 1, r);
    node[rt].sum = node[rt << 1].sum + node[(rt << 1) | 1].sum;
}
void update(int rt, int pos, int v)
{
    if(node[rt].l == pos && node[rt].r == pos)
    {
        node[rt].sum += v;
        return;
    }
    int mid = node[rt].getmid();
    if (pos > mid)
        update((rt << 1) | 1, pos, v);
    else
        update(rt << 1, pos, v);
    node[rt].sum = node[rt << 1].sum + node[(rt << 1) | 1].sum;
}

int GetSum(int rt, int l, int r)
{
    if(node[rt].l == l && node[rt].r == r)
        return node[rt].sum;
    int mid = node[rt].getmid();
    if (l > mid)
        return GetSum((rt << 1) | 1, l, r);
    else if (r < mid)
        return GetSum(rt << 1, l, r);
    else
        return GetSum(rt << 1, l, mid) + GetSum((rt << 1) | 1, mid + 1, r);
}

struct Tree
{
    int l, r;
    int dat, lazy;
    Tree() { lazy = 0; }
} node[maxn << 2];
void build(int p, int l, int r)
{
    node[p].l = l;
    node[p].r = r;
    if (l == r)
    {
        node[p].dat = 0;
        return;
    }

    int m = l + ((r - l) >> 1);
    build(p << 1, l, m);
    build(p << 1 | 1, m + 1, r);
    node[p].dat = node[p << 1 | 1].dat + node[p << 1].dat;

    return;
}
void spread(int p)
{
    if (node[p].lazy)
    {
        node[p << 1].dat += node[p].lazy * (node[p << 1].r - node[p << 1].l + 1);
        node[p << 1 | 1].dat += node[p].lazy * (node[p << 1 | 1].r - node[p << 1 | 1].l + 1);
        node[p << 1].lazy += node[p].lazy;
        node[p << 1 | 1].lazy += node[p].lazy;
        node[p].lazy = 0;
    }
    return;
}
void update(int p, int val, int l, int r)
{
    if (l <= node[p].l && r >= node[p].r)
    {
        node[p].lazy += val;
        node[p].dat += val * (node[p].r - node[p].l + 1);
        return;
    }
    spread(p);
    int m = node[p].l + ((node[p].r - node[p].l >> 1));
    if (l <= m)
        update(p << 1, val, l, r);
    if (r > m)
        update(p << 1 | 1, val, l, r);

    node[p].dat = node[p << 1 | 1].dat + node[p << 1].dat;

    return;
}

int query(int p, int l, int r)
{
    if (l <= node[p].l && node[p].r <= r)
    {
        return node[p].dat;
    }
    spread(p);
    int ans = 0;
    int mid = node[p].l + ((node[p].r - node[p].l >> 1));
    if (l <= mid)
        ans += query(p << 1, l, r);
    if (r > mid)
        ans += query(p << 1 | 1, l, r);
    return ans;
}

class djset{
public:
    vector<int> parent;
    vector<int> rank;
    djset(int n):parent(vector<int>(n)),rank(vector<int>(n)){
        for (int i = 0; i < n;i++)
            parent[i] = i;
    }
    int find(int x)
    {
        if(x != parent[x])
            parent[x] = find(parent[x]);
        return parent[x];
    }
    void merge(int x,int y)
    {
        int a = find(x), b = find(y);
        if(a==b)
            return;
        if(rank[a]<rank[b])
            parent[a] = parent[b];
        else
        {
            parent[b] = parent[a];
            if(rank[a]==rank[b])
                rank[a]++;
        }
    }
};

8. RMQ 算法(ST表)

/*
 * st表
 * st[i][j]表示元素arr[i]开始，长度为2^j区间内的最值
 * st[i][0]为arr[i]，st[i][j] = max(st[i][j-1], st[i - (1 << j-1)][j-1])
 * 求区间[l, r]内最值：
 * Log[i]代表i的对数向下取整
 * 对于长度len而言，显然2^log2[len]严格大于len的一半
 * 令k = Log[r - l + 1]，则最值为max(st[l][k], st[r - (1<<k) + 1][k])
 */
#define maxn (55000)
int Log[maxn], st[maxn][32];
void st_prepare(int n, int *arr)
{
     Log[1] = 0;
    for(int i = 2; i <= n; i++)
    {
        Log[i]=Log[i>>1]+1;
        st[i][0] = arr[i];
    }
    st[1][0] = arr[1];
    for (int j = 1; (1<<j) <= n; j++)
        for (int i = 1; i + (1 << j) - 1 <= n; i++)
            st[i][j] = min(st[i][j - 1], st[i + (1 << (j - 1))][j - 1]);
}
int st_query(int l, int r)
{
    int len = r - l + 1, k = Log[len];
    return max(st[l][k], st[r - (1 << k) + 1][k]);
}

class KMP{
public:
    vector<int> next;
    void Get_Next(const string &p)
    {
        int n = p.size();
        next.resize(n + 1, 0);
        int i = 0, j = -1;
        next[0] = -1;
        while (i < n)
        {
            if(j == -1 || p[i]==p[j])
            {
                i++, j++;
                next[i] = j;
            }
            else
                j = next[j];
        }
    }
    //p:匹配串 ； t: 原串
    int Kmp(const string &t, const string &p)
    {
        Get_Next(p);
        int i = 0, j = 0;
        int nt = t.size(), np = p.size();
        while (i < nt && j < np)
        {
            if (j == -1 || t[i] == p[j])
                i++, j++;
            else
                j = next[j];
        }
        if(j == np)
            return i - j;
        return -1;
    }
};

void manacher(string& s, int *R, int n)
{
/*
 * manacher算法
 * 需要将字符串预处理成$#x#x#x#x#x#x#形式 ✔
 * 若仅求长度为奇数的回文串，最左侧添加特殊字符即可
 * 记录当前最右延伸回文半径mx和对应回文中心p
 * i若位于mx以内，则将对称位置2*p-i的回文半径的不越界部分作为i的回文半径，并且继续向右侧匹配
 * 若得到新的最右延伸回文半径，更新mx和p
 * 回文长度为回文半径-1
 * 回文起始位置为 （回文中心位置-回文半径）/2
 */
    int p = 0, mx = 0;
    R[0] = 1;
    for(int i = 1; i < n; i++)
    {
        if(mx > i)
            R[i] = min(R[2*p - i], mx - i);
        else R[i] = 1;
        while(s[i - R[i]] == s[i + R[i]])
            R[i]++;
        if(i + R[i] > mx)
            p = i, mx = i + R[i];
    }
    return;
}

	{ //新建节点
		for (int i = 0; i < N; ++i)
			next[p][i] = 0;
		cnt[p] = 0;
		id[p] = 0;
		num[p] = 0;
		len[p] = l;
		return p++;
	}

	{ //初始化
		p = 0;
		newnode(0);
		newnode(-1);
		last = 0;
		n = 0;
		S[n] = -1; //开头放一个字符集中没有的字符，减少特判
		fail[0] = 1;
	}

	{ //和KMP一样，失配后找一个尽量最长的
		while (S[n - len[x] - 1] != S[n])
			x = fail[x];
		return x;
	}

	{
		c -= 'a';
		S[++n] = c;
		int cur = get_fail(last); //通过上一个回文串找这个回文串的匹配位置
		if (!next[cur][c])
		{											  //如果这个回文串没有出现过，说明出现了一个新的本质不同的回文串
			int now = newnode(len[cur] + 2);		  //新建节点
			fail[now] = next[get_fail(fail[cur])][c]; //和AC自动机一样建立fail指针，以便失配后跳转
			next[cur][c] = now;
			num[now] = num[fail[now]] + 1;
		}
		last = next[cur][c];
		cnt[last]++;
		id[last] = n; //id[第last个节点]=第n个字符
	}

	{
		int le = strlen(s), tmp;
		for (int i = 0; i < le; i++)
		{
			tmp = s[i] - 'a';
			add(tmp);
		}
	}

	{
		for (int i = p - 1; i >= 0; --i)
			cnt[fail[i]] += cnt[i];
		//父亲累加儿子的cnt，因为如果fail[v]=u，则u一定是v的子回文串！
	}

    {
        edges.push_back((Edge){from,to,cap,0});
        edges.push_back((Edge){to,from,0,0});
        int m=edges.size();
        G[from].push_back(m-2);
        G[to].push_back(m-1);
    }

    {
        int x=t,a=INF;
        while(x!=s)//找最小的残量值
        {
            Edge&e=edges[p[x]];
            a=min(a,e.cap-e.flow);
            x=edges[p[x]].from;
        }
        x=t;
        while(x!=s)//增广
        {
            edges[p[x]].flow+=a;
            edges[p[x]^1].flow-=a;
            x=edges[p[x]].from;
        }
        return a;
    }
    void bfs()//逆向进行bfs
    {
        memset(vis,0,sizeof(vis));
        queue<int>q;
        q.push(t);
        d[t]=0;
        vis[t]=1;
        while(!q.empty())
        {
            int x=q.front();q.pop();
            int len=G[x].size();
            for(int i=0;i<len;i++)
            {
                Edge&e=edges[G[x][i]];
                if(!vis[e.from]&&e.cap>e.flow)
                {
                    vis[e.from]=1;
                    d[e.from]=d[x]+1;
                    q.push(e.from);
                }
            }
        }
    }
    int Maxflow(int s,int t)//根据情况前进或者后退，走到汇点时增广
    {
        this->s=s;
        this->t=t;
        int flow=0;
        bfs();
        memset(num,0,sizeof(num));
        for(int i=0;i<n;i++)
            num[d[i]]++;
        int x=s;
        memset(cur,0,sizeof(cur));
        while(d[s]<n)
        {
            if(x==t)//走到了汇点，进行增广
            {
                flow+=Augumemt();
                x=s;//增广后回到源点
            }
            int ok=0;
            for(int i=cur[x];i<G[x].size();i++)
            {
                Edge&e=edges[G[x][i]];
                if(e.cap>e.flow&&d[x]==d[e.to]+1)
                {
                    ok=1;
                    p[e.to]=G[x][i];//记录来的时候走的边，即父边
                    cur[x]=i;
                    x=e.to;//前进
                    break;
                }
            }
            if(!ok)//走不动了，撤退
            {
                int m=n-1;//如果没有弧，那么m+1就是n，即d[i]=n
                for(int i=0;i<G[x].size();i++)
                {
                    Edge&e=edges[G[x][i]];
                    if(e.cap>e.flow)
                        m=min(m,d[e.to]);
                }
                if(--num[d[x]]==0)break;//如果走不动了，且这个距离值原来只有一个，那么s-t不连通，这就是所谓的“gap优化”
                num[d[x]=m+1]++;
                cur[x]=0;
                if(x!=s)
                    x=edges[p[x]].from;//退一步，沿着父边返回
            }
        }
        return flow;
    }

{

 * 后缀数组的倍增构造法
 * 复杂度为O(nlogn)
 * 
 * 从 sa[i] 开始得后缀排在i个
 * 从 i 开始得后缀排名为 rank[i]

const int maxn=1e6+10;
bool cmp(int *r, int a, int b, int l)
{ return r[a] == r[b] && r[a + l] == r[b + l]; }
int ta[maxn], tb[maxn], bk[maxn];
void da(int *r, int *sa, int n, int m)
{
    int i, j, p, *x = ta, *y = tb, *t;
    for(i = 0; i < m; i++) bk[i] = 0;
    for(i = 0; i < n; i++) bk[x[i] = r[i]]++;
    for(i = 1; i < m; i++) bk[i] += bk[i-1];
    for(i = 0; i < n; i++) sa[--bk[x[i]]] = i;
    for(j = 1, p = 1; p < n; j <<=1, m = p)
    {
        for(p = 0, i = n - j; i < n; i++) y[p++] = i;
        for(i = 0; i < n; i++) if(sa[i] >= j) y[p++] = sa[i] - j;
        for(i = 0; i < m; i++) bk[i] = 0;
        for(i = 0; i < n; i++) bk[x[i]]++;
        for(i = 1; i < m; i++) bk[i] += bk[i-1];
        for(i = n-1; i >= 0; i--) sa[--bk[x[y[i]]]] = y[i];
        for(t = x, x = y, y = t, x[sa[0]] = 0, p = 1, i = 1; i < n; i++)
            x[sa[i]] = cmp(y, sa[i-1], sa[i], j)? p-1: p++;
    }
}

class __SA
{

    {
        return r[a] == r[b] && r[a + l] == r[b + l];
    }
    template <typename S>
    void _sa(S &&s, uint32_t n, uint32_t m)
    {
        ++n;
        sa = make_unique<T[]>(n);
        rk = make_unique<T[]>(n);
        ht = make_unique<T[]>(n);
        auto &&x = rk;
        auto &&y = ht;
        const auto wv = make_unique<T[]>(n);
        const auto ws = make_unique<T[]>(max(m, n));
        int i, j, p, k = 0;
        for (i = 0; i < m; i++)
            ws[i] = 0;
        for (i = 0; i < n; i++)
            ws[x[i] = s[i]]++;
        for (i = 1; i < m; i++)
            ws[i] += ws[i - 1];
        for (i = n - 1; i >= 0; i--)
            sa[--ws[x[i]]] = i;
        for (j = 1, p = 1; p < n; j *= 2, m = p)
        {
            for (p = 0, i = n - j; i < n; i++)
                y[p++] = i;
            for (i = 0; i < n; i++)
                if (sa[i] >= j)
                    y[p++] = sa[i] - j;
            for (i = 0; i < n; i++)
                wv[i] = x[y[i]];
            for (i = 0; i < m; i++)
                ws[i] = 0;
            for (i = 0; i < n; i++)
                ws[wv[i]]++;
            for (i = 1; i < m; i++)
                ws[i] += ws[i - 1];
            for (i = n - 1; i >= 0; i--)
                sa[--ws[wv[i]]] = y[i];
            for (x.swap(y), p = 1, x[sa[0]] = 0, i = 1; i < n; i++)
                x[sa[i]] = cmp(y, sa[i - 1], sa[i], j) ? p - 1 : p++;
        }
    }

public:
    std::unique_ptr<T[]> sa, rk, ht;
    template <typename S>
    __SA(S &&s, uint32_t n, uint32_t m = 128)
    {
        _sa(s, n, m);
        int i, j, k = 0;
        for (i = 1; i <= n; i++)
            rk[sa[i]] = i;
        for (i = 0; i < n; ht[rk[i++]] = k)
            for ((k ? k-- : 0), j = sa[rk[i] - 1]; s[i + k] == s[j + k]; k++)
                ;
    }
    int get_sa(int idx) { return sa[idx + 1]; }
    int get_rk(int idx) { return rk[idx] - 1; }
    int get_ht(int idx) { return ht[idx + 1]; }

const int maxn = 5e5 + 10;
const int inf = 0x7ffffff;

struct node
{
    int to, w, next;
} edge[maxn << 1];
int head[maxn], dis[maxn];
int tot = 0;
void addedge(int x, int y, int z)
{
    tot++;
    edge[tot].w = z, edge[tot].to = y, edge[tot].next = head[x];
    head[x] = tot;
}

int depth[maxn], fa[maxn][22], lg[maxn], dis[maxn];

void prepare(int n)
{
    lg[1] = 1;
    lg[0] = 0;
    for (int i = 2; i <= n; i++)
        lg[i] = lg[i >> 1] + 1;
}

void dfs(int now, int father)
{
    fa[now][0] = father, depth[now] = depth[father] + 1;
    for (int i = 1; i <= lg[depth[now]]; i++)
        fa[now][i] = fa[fa[now][i - 1]][i - 1];
    for (int i = head[now]; i; i = edge[i].next)
        if (edge[i].to != father)
        {
            dis[edge[i].to] = dis[now] + edge[i].w;
            dfs(edge[i].to, now);
        }
}

int LCA(int x, int y)
{
    if (depth[x] < depth[y])
        swap(x, y);
    while (depth[x] > depth[y])
        x = fa[x][lg[depth[x] - depth[y]] - 1];
    if (x == y)
        return x;
    for (int k = lg[depth[x]] - 1; k >= 0; --k)
        if (fa[x][k] != fa[y][k])
            x = fa[x][k], y = fa[y][k];
    return fa[x][0];
}

    int x, y;
    for (int i = 1; i < N;i++)

    }

    dfs(S, 0);

}
