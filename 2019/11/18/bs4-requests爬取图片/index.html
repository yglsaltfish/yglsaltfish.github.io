import requests
from bs4 import BeautifulSoup
import time

def geturl(url):
    r=requests.get(url)
    soup=BeautifulSoup(r.text,'lxml')
    image_url=soup.find_all('ul',class_='clearfix')
    target_url=image_url[1].find_all('a',target='_blank')
    return target_url
    
def getcontent(url,index):
