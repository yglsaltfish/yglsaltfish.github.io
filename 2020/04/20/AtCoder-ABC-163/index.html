AtCoder ABC 163 (D~F)

D.Sum of Large Numbers

  • 题意：
    给定两个数，$N,K$ ，有$N+1$个数，分别是$10^{100}+1,10^{100}+2…,10^{100}+N$,问至少取K个数，取得的数字之和不同的个数是多少。
  • 题解:
    可以发现，取不同个数的数字是不同的，所以只要知道在${0,1,2,…,N} , N+1$个数中,取x个数，这x个数表示的范围是多少。可以发现$1+2+…+x \leq sum\leq(n-x)+(n-x+1)+…+n$所以，知道了这些，就可以枚举$k$到$n$，就能得出答案了。

int n, k;
void solve()
{
    cin >> n >> k;
    ll lsum = 0, rsum = 0;
    for (int i = 0; i < k;i++)
    {
        lsum += i;
        rsum += n  - i;
    }
    ll sum = 0;
    sum = (sum + rsum - lsum + 1) % mod;
    for (int i = k; i < n + 1;i++)
    {
        lsum += i, rsum += n - i;
      //  cout << lsum << " " << rsum << " " << rsum - lsum << "\n";
        sum = (sum + rsum - lsum + 1) % mod;
    }
    cout << sum << "\n";
}

E. Active Infants

  • 题意：
    给定$N$个数，可以进行一次任意调序。定义一个$sum$，表示$\sum_{i=1}^{n} a[i]*abs(pos_{change})$，问求得的$sum$最大是多少？
  • 题解：
    可以发现，使得当前值最大的该变量就是将当前值移到数组首端或者末端。就可以很自然而然的想到要用dp求解。
    $dp[l][r]$ 表示在$l~r$这个范围还没有被占用的最大$sum$值。
    可以得到状态转移方程:
    $dp[l][r] = max(dp[l][r], dp[l][r + 1] + a[i]* abs(pos - (r+1));
    dp[l][r] = max(dp[l][r], dp[l-1][r] + a[i]* abs(pos - (l-1));$
ll dp[maxn][maxn];

void solve()
{
    int N;
    cin >> N;
    vector<pair<ll, ll>> vec(N);
    for (int i = 0; i < N;i++)
    {
        cin >> vec[i].first;
        vec[i].second = (ll)(i + 1);
    }
    sort(vec.rbegin(), vec.rend());
    for (int i = 1; i <= N;i++)
    {
        for (int j = 0; j <= i;j++)
        {
            int l = j + 1, r = N - i + j;
            if(r + 1 <= N)
                dp[l][r] = max(dp[l][r], dp[l][r + 1] + vec[i - 1].first * abs(vec[i - 1].second - r - 1));
            if(l - 1 >= 1)
                dp[l][r] = max(dp[l][r], dp[l - 1][r] + vec[i - 1].first * abs(vec[i - 1].second - l + 1));
        }
    }
    ll ans = 0;
    for (int i = 1; i <= N;i++)
    {
        ans = max(ans, dp[i][i] + vec[N - 1].first * abs(i - vec[N - 1].second));
    }
    cout << ans << "\n";
}

F. path pass i

  • 题意：
    给定一棵树，有$N$个节点，每个节点有一种颜色，问包含每种颜色的简单路径个数是多少。

  • 题解：
    正的计数很难做，可以知道经过所有节点的简单路径总个数是$N*(N+1)/2$ 。那么只要求出简单路径上没有该颜色的个数，就可以求出答案。

ll cal(ll n) { return n * (n + 1) / 2; }
vector<int> G[maxn]; //图
int c[maxn];    //颜色

int sz[maxn], sum[maxn];//sz[i]表示以i为根节点的子节点个数 ,  sum[i] 表示 以颜色为 i 的所有节点个数
ll ans[maxn]; //答案
void dfs(int u,int fa)
{
    sz[u] = 1;
    int in = sum[c[u]];
    for(int v:G[u])
    {
        if(v == fa)
            continue;
        int front = sum[c[u]];
        dfs(v, u);
        int back = sum[c[u]] - front;
        ans[c[u]] -= cal((ll)sz[v] - (ll)back);
        sz[u] += sz[v];
    }
    
    sum[c[u]] = sz[u] + in;
}
void solve()
{
    int N;
    int a, b;
    cin >> N;
    for (int i = 1; i <= N;i++)
        cin >> c[i];
    for (int i = 1; i < N;i++)
    {
        cin >> a >> b;
        G[a].push_back(b);
        G[b].push_back(a);
        ans[i] = cal((ll)N);
    }
    ans[N] = cal((ll)N);
    dfs(1, 1);
    for (int i = 1; i <= N;i++)
    {
        ans[i] -= cal((ll)(N - sum[i]));
        cout << ans[i] << "\n";
       // cout << c[i]<<" "<<sum[i] << "\n";
    }
}
