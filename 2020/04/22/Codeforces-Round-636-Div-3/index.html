B. Balanced Array

  • 题意: 给定一个偶数\(n\) , 输出 \(2n\) 个数，前\(n\)个数为偶数，后\(n\)个数为奇数，要求偶数的和==奇数的和。如果存在就输出，不存在就输出"NO"。
  • 题解&代码 可以通过奇偶性质判断，当\(\frac{n}{2}\)为奇数时，奇数的和为奇数，则无法满足要求。为偶数时按题意模拟即可. 
    void solve()
{
    int N;
    cin >> N;
    if(N/2&1)
    {
        cout << "NO\n";
        return;
    }
    cout << "YES\n";
    ll sum = 0;
    ll tmp = 0;
    for (int i = 1; i <= N / 2;i++)
    {
        tmp+=2;
        cout << tmp << " ";
        sum += tmp;
    }
    tmp = -1;
    for (int i = 1; i < N / 2;i++)
    {
        tmp += 2;
        cout << tmp << " ";
        sum -= tmp;
    }
    cout << sum << "\n";
}

C. Alternating Subsequence

  • 题意: 给定一个包含\(n\)个数的序列，让你求出最长的子序列，其中子序列种的相邻两个数必须不同号，输出子序列的和。如果相同长度，则输出最大子序列的和。

  • 题解&代码 划分区间，每个区间取最大值即可。

ll a[maxn];
void solve()
{
    int N;
    cin >> N;
    for (int i = 1; i <= N;i++)
        cin >> a[i];
    ll sum = 0;
    ll tmp = a[1];
    ll lst = a[1];
    for (int i = 2; i <= N + 1; i++)
    {
        if(lst*a[i]<0|| i==N+1)
        {
            sum += tmp;
            lst = a[i];
            tmp = a[i];
        }
        tmp = max(tmp, a[i]);
    }
    cout << sum << "\n";
}

D. Constant Palindrome Sum

  • 题意: 给定偶数\(n\), 正整数\(k\)，和包含\(2n\)个数的数组\(a[2n]\), 可以将数组中的数改变为\([1:k]\) 这个范围的任意数使得\(a[i]+a[n-i+1]\) 之和全部相等，改变数字的最小次数是多少。

  • 题解&代码: 用线段树或者差分都能做。我比赛的时候线段树写崩了。。。QAQ，这里讲差分的做法。 \(a[i]+a[n-i+1]\) 改变一个数字所能达到的最大范围是 \([min(a[i],a[n-i+1])+1,max(a[i],a[n-i+1])+k]\),那么我们只要把这个区间的数+1，然后就可以发现没有+1的区间需要改变两次才能成功。那么我们只要枚举\([1:2k]\)，更新答案即可。


int pref[maxn];
int a[maxn];
int vis[maxn];
void solve()
{
    int n, k;
    cin >> n >> k;
    fill(vis, vis + 2 * k + 1, 0);
    fill(pref, pref + 2 * k + 1, 0);
    for (int i = 1; i <= n;i++)
        cin >> a[i];
    for (int i = 1; i <= n / 2;i++)
    {
        int tmp = a[i] + a[n - i + 1];
        vis[tmp]++;
    }
    for (int i = 1; i <= n / 2; i++)
    {
        int l = 1 + a[i], r = k + a[i];
        int tl = 1 + a[n - i + 1], tr = k + a[n - i + 1];
        int mn = min(l, tl), mx = max(r, tr);
        pref[mn]++, pref[mx+1]--;
    }
    for (int i = 1; i <= k * 2+1;i++)
        pref[i] += pref[i - 1];
    int ans = inf;
    for (int i = 2; i <= 2 * k;i++)
    {
        ans = min(ans, pref[i] - vis[i] + (n / 2 - pref[i]) * 2);
    }
    cout << ans << "\n";
}

E. Weights Distributing

  • 题意： 给定\(n\) 个点，\(m\)条边，\(m\)个边的权值，点\(a,b,c\),让你求出如何分配权值，使得\(a->b,b->c\)这条路径的权值和最小，输出这个权值和。

  • 题解&代码: 考虑中间点\(x\),那么只要求出\(a->x,x->b,b->x,c->x\)之和即可，使用三次bfs求出\(a,b,c\)到各点的距离，之后枚举点即可。

const int maxn = 5e5 + 10;
const int inf = 0x7ffffff;
int n, m, a, b, c;
int p[maxn];
vector<int> G[maxn];
int vis[maxn];
int pre[maxn];
void bfs(int rt, vector<int> &vec)
{
    queue<int> que;
    que.push(rt), vec[rt] = 0;
    int tp;
    while (!que.empty())
    {
        tp = que.front(), que.pop();
        for (int i : G[tp])
        {
            if (vec[i]==inf)
            {
                vec[i] = vec[tp] + 1;
                que.push(i);
            }
        }
    }
}
ll pref[maxn];
void solve()
{
    cin >> n >> m >> a >> b >> c;
    fill(pref, pref + m + 1, 0);
    for (int i = 0; i <= n;i++)
        G[i].clear();
    for (int i = 1; i <= m; i++)
        cin >> p[i];
    int u, v;
    for (int i = 1; i <= m; i++)
    {
        cin >> u >> v;
        G[u].push_back(v), G[v].push_back(u);
    }
    vector<int> path1(n + 1,inf), path2(n + 1,inf), path3(n + 1,inf);
    bfs(a, path1), bfs(b, path2), bfs(c, path3);
    sort(p + 1, p + m + 1);
    for (int i = 1; i <= m;i++)
        pref[i] = pref[i - 1] + (ll)p[i];
    ll ans = 1e18;
    for (int i = 1; i <= n;i++)
    {
        if(path1[i] + path2[i] + path3[i]>m)
            continue;
        ans = min(ans, pref[path2[i]] + pref[path1[i] + path2[i] + path3[i]]);
    }
        
    cout << ans << "\n";
}
