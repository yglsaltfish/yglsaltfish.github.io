C. Many Requirements

  • 题意:
    给定 $N,M,Q$ 表示有一个长度为 $N$ 的数组 $A$，数组的每个元素的范围为 $1 \leq A[i] \leq M$, 然后给定 $Q$ 个四元组 $a_i,b_i,c_i,d_i$ ，如果满足 $A[b_i]-A[a_i]=c_i$,则该数组的价值 $+d_i$ , 问价值最高是多少，数组 $A$ 为不下降序列。

  • 题解：
    一开始以为 $dfs$ 的话会超时，后来算了一下，发现只要枚举 $\tbinom{N+M+1}{M-1}$ 即可，这样最大值就是 $\tbinom{19}{9}=92378$ , $dfs$ 并不会超时。

struct nd
{
    int a, b, c, d;
}node[60];
int n, m, q;
int a[20];

ll sum = 0;
void calc()
{
    ll tmp = 0;
    for (int i = 1; i <= q;i++)
    {
        if(a[node[i].b]-a[node[i].a]==node[i].c)
            tmp += node[i].d;
    }
    sum = max(sum, tmp);
}

void dfs(int pos)
{
    if(pos==n+1)
    {
        calc();
        return;
    }
    for (int i = a[pos-1]; i <= m;i++)
    {
        a[pos] = i;
        dfs(pos + 1);
    }
}

void solve()
{
    a[0] = 1;
    cin >> n >> m >> q;
    for (int i = 1; i <= q;i++)
        cin >> node[i].a >> node[i].b >> node[i].c >> node[i].d;
    dfs(1);
    cout << sum << "\n";
}

D. Floor Function

  • 题意:
    给定 $A,B,N$，让你求出一个数 $x$ 使得 $floor(Ax/B)-A*floor(x/B)$的最大值

  • 题解:
    发现 $x=min(B-1,n)$ 的时候取最大值。

void solve()
{
    db a, b, n;
    cin >> a >> b >> n;
    db x = min(b - 1, n);
    cout << floor(a * x / b) << "\n";
}

E. Rotation Matching

  • 题意:
    $N$ 个人，$M$ 个竞技场，每个竞技场会有编号，对应编号的人参与竞技，一次竞技之后所有参赛人数的编号 $+1 mod N$ ,问每个对手不不重复对战两次的排序方式。

  • 题解:
    相差放数字即可。

 void solve()
{
    int n, m;
    cin >> n >> m;
    int pos = 1;
    int x = (n - 1) / 2;
    int f = 0, l = 1, r = n;
    for (int i = 1; i <= m;i++)
    {
        if(f)
            cout << l << " " << l + x << "\n", l++, x--;
        else
            cout << r << " " << r - x << "\n", r--, x--;
        f ^= 1;
    }
}

F. LIS on Tree

  • 题意:
    给一个有 $N$ 个节点的树，每个节点有一个权值，问从 $1$ 开始访问到第 $k$ 个节点经过的路径权值序列的最长上升序列的长度是多少。

  • 题解：
    可以一边 $dfs$ 一边跑$LIS$，然后记录答案即可。

 vector<int> G[maxn], rd;
int N, a[maxn];
int ans[maxn];
void dfs(int u,int p)
{
    int now = lower_bound(rd.begin(), rd.end(), a[u]) - rd.begin();
    int pop = 0, lst;
    if(now==rd.size())
        pop = 1, rd.push_back(a[u]);
    else
        lst = rd[now], rd[now] = a[u];
    ans[u] = rd.size();
    for(int v:G[u])
        if(v!=p)
            dfs(v, u);
    if(pop)
        rd.pop_back();
    else
        rd[now] = lst;

}
void solve()
{
    int u, v;
    cin >> N;
    for (int i = 1; i <= N;i++)
        cin >> a[i];
    for (int i = 1; i < N;i++)
    {
        cin >> u >> v;
        G[u].push_back(v), G[v].push_back(u);
    }
    dfs(1, 0);
    for (int i = 1; i <= N;i++)
        cout << ans[i] << "\n";
}
