struct nd
{
    int a, b, c, d;
}node[60];
int n, m, q;
int a[20];

ll sum = 0;
void calc()
{
    ll tmp = 0;
    for (int i = 1; i <= q;i++)
    {
        if(a[node[i].b]-a[node[i].a]==node[i].c)
            tmp += node[i].d;
    }
    sum = max(sum, tmp);
}

void dfs(int pos)
{
    if(pos==n+1)
    {
        calc();
        return;
    }
    for (int i = a[pos-1]; i <= m;i++)
    {
        a[pos] = i;
        dfs(pos + 1);
    }
}

void solve()
{
    a[0] = 1;
    cin >> n >> m >> q;
    for (int i = 1; i <= q;i++)
        cin >> node[i].a >> node[i].b >> node[i].c >> node[i].d;
    dfs(1);
    cout << sum << "\n";
}

void dfs(int u,int p)
{
    int now = lower_bound(rd.begin(), rd.end(), a[u]) - rd.begin();
