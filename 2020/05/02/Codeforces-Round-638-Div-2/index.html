A. Phoenix and Balance

  • 题意:
    给定一个数 $n$ ，让你将$2^1 + 2^2 + …+2^n$ 等分成两组数，使得这两组数的绝对值之差最小，输出最小值。
  • 题解:
    会发现可以用二进制表示这 $n$ 个数，那么对于给定$n=6$,则可以表示为 $100011$ 和$011100$，这两个数的绝对值之差最小。按这个思路模拟即可。
void solve()
{
    ll  n;
    cin >> n;
    ll sum = 0;
    sum = (1 << (n + 1)) - 2;
    int cnt = 0;
    for (ll i = n-1;; i --)
    {
        if(cnt==n/2)
            break;
        sum -= (1 << (i + 1));
        cnt++;
    }
    cout << abs(sum) << "\n";
}

B.Phoenix and Beauty

  • 题意:
    给定两个数 $n,k$ ,和一个包含$n$个数的序列 , 通过向原串中插入数字构造一个串 , 使得任意长度为k的子串之和相等.

  • 题解:

  1. 发现如果原序列的数字种类如果大于 $k$ 的话,无法构成序列.
  2. 如果小于 $k$ 的话,则可以填充任意数字到 $k$ ,然后每个序列重复$n$次即可.
int a;
void solve()
{
    set<int> st;
    int n, m;
    cin >> n >> m;
    for (int i = 1; i <= n;i++)
    {
        cin>>a;
        st.insert(a);
    }
    vector<int> vec(st.begin(), st.end());
    if(st.size()<=m)
    {
        while ((int)vec.size()<m)
            vec.push_back(vec[0]);
        cout << n * (int)vec.size() << "\n";
        for (int i = 0; i < n;i++)
        {
            for (int j = 0; j < vec.size();j++)
                cout << vec[j] << " ";
        }
        cout << "\n";
    }
    else
        cout << -1 << "\n";
}

C. Phoenix and Distribution

  • 题意:
    给定两个数字 $n,k$ ,和包含n个字母的字符串,让你讲这个串以任意序列分割成 $k$ 份,定义一个 $max$ 表示分割完成后字典序最大的字符串,求出最小字符串 $max$.

  • 题解:
    先排序,然后将k个字符串多分配一个字母,以确保非空.之后看剩下的字母种类,如果只剩下一种字母,则循环分配 ; 如果剩下的字母大于一种,则将所有串都分配到字典序最小的那个串上.之后将分好的串排序,即可得到答案.

void solve()
{
    string s;
    int n, k;
    cin >> n >> k >> s;
    sort(s.begin(), s.end());
    vector<string> vec(k);
    int pos = 0, fla = 1;
    for (int i = 0; i < k;i++)
    {
        vec[i].push_back(s[i]);
        if (i != k - 1 && s[i] == s[i + 1] &&fla)
            pos++;
        else
            fla = 0;
    }
    pos++;
  //  cout << pos << "\n";
    set<char> st;
    for (int i = k; i < (int)s.size();i++)
        st.insert(s[i]);
    if((int)st.size()==1)
    {
        int tmp = 0, i=k;
        while (i<(int)s.size())
        {
            vec[tmp].push_back(s[i]);
            i++;
            tmp++;
            tmp %= pos;
        }
        sort(vec.rbegin(), vec.rend());
        cout << vec[0] << "\n";
    }
    else
    {
        for (int i = k; i < (int)s.size();i++)
            vec[0].push_back(s[i]);
        sort(vec.rbegin(), vec.rend());
        cout << vec[0] << "\n";
    }
}

E. Phoenix and Berries

  • 题意:
    给 $n$ 个灌木丛和 容积为 $k$ 的袋子,n个灌木丛分别由 $a_i$ 个红浆果和 $b_i$ 个红浆果. 同种颜色或者同一个浆果丛的浆果可以放到同一个袋子里.求出能放满的袋子的最大数量.

  • 题解:
    发现对于所有浆果而言,能放满的袋子数量最大为 $max((suma+sumb)/k , suma+sumb)/k-1)$ .
    我们可以使用dp求解. $dp[i][j]$ 表示到第 $i$ 个灌木丛,剩下 $j$ 个红浆果能否达到.

int a[maxn], b[maxn];
int dp[maxn][maxn];
void solve()
{
    ll suma = 0, sumb = 0;
    int n, k;
    cin >> n >> k;
    for (int i = 1; i <= n;i++)
    {
        cin >> a[i] >> b[i];
        suma += a[i], sumb += b[i];
    }
    dp[0][0] = 1;
    for (int i = 1; i <= n;i++)
    {
        for (int j = 0; j < k;j++)
        {
            dp[i][j] |= dp[i - 1][(j - a[i] % k + k) % k]; //当前红浆果的
            for (int x = 0; x <= min(k - 1, a[i]);x++)     
            {
                if((a[i]-x)%k+b[i]>=k)          //使用同一种灌木丛的蓝浆果是否能满足
                {
                    dp[i][j] |= dp[i - 1][(j - x + k) % k];
                }
            }
        }
    }
    ll ans = 0;
    for (int i = 0; i <= k;i++)
        if(dp[n][i])
            ans = max(ans, (suma + sumb - i) / k);
    cout << ans << "\n";
}
