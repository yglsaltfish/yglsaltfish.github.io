A. Suffix Three

  • 题意：
    给定一个字符串，结尾有标识符，让你输出其类型。
  • 题解&代码：
    直接按照题意模拟即可。
#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <queue>
#include <vector>

using namespace std;

#define ll long long
#define db double
#define clr(a, b) memset(a, b, sizeof(a))
#define vi vector<int>

const int maxn = 5e5 + 10;

int n, t;
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> n;
    int ans;
    while (n--)
    {
        string s;
        cin >> s;
        int len = s.size();
        if (s[len - 2] == 'p')
            ans = 1;
        else if (s[len - 4] == 'd' || s[len - 4] == 'm')
            ans = 2;
        else
            ans == 3;

        if (ans == 1)
            cout << "FILIPINO" << endl;
        else if (ans == 2)
            cout << "JAPANESE" << endl;
        else
            cout << "KOREAN" << endl;
        ans = 0;
    }
    return 0;
}

B. Azamon Web Services

  • 题意：
    给定两个字符串A，B，问最多交换1次A中的两个字符，然后使得A的字典序大于B的字典序，若能则输出A的字符。
  • 题解&代码：
    贪心求解。假设$s_{pos}$ 是A中的一个字符，只要使得下标$pos$最小，选择$s_{pos}$ 以后的最小的字典序的字符，并且交换即可。
int n, t;

string s1, s2;
struct node
{
    char x;
    int pos;
} vec[maxn];
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> n;
    while (n--)
    {
        cin >> s1 >> s2;
        int len = s1.length();
        for (int i = 0; i < len; i++)
            vec[i].x = vec[i].pos = 0;
        vec[len - 1].x = s1[len - 1];
        vec[len - 1].pos = len - 1;
        for (int i = len - 2; i >= 0; i--)
        {
            if (vec[i + 1].x > s1[i])
            {
                vec[i].x = s1[i];
                vec[i].pos = i;
            }
            else
            {
                vec[i].x = vec[i + 1].x;
                vec[i].pos = vec[i + 1].pos;
            }
        }
        for (int i = 0; i < len - 1; i++)
        {
            if (s1[i] > vec[i + 1].x)
            {
                int len = vec[i + 1].pos;
                swap(s1[i], s1[len]);
                break;
            }
        }
        if (s1 < s2)
            cout << s1 << endl;
        else
            cout << "---" << endl;
    }
    return 0;
}

C. Cut and Paste

  • 题意：
    给定一个数$n$ 和一段由 $1,2,3$ 组成的字符串$s$ 。给定一个数$l=0$ ，进行如下操作:

    1、$l++$

    2、将$s_{l}$ 右边的（包括$s_l$ 本身）复制$s_{l}-1$ 次，并添加到字符串$s$ 的末端。

    3、重复操作1，直到$l=n$

  • 题解&代码：
    模拟+数学。因为数据范围$n<10^6$ ,所以考虑分数据范围写法，先按照题目要求将字符串$s$ 的长度增长到大于$n$ 的长度，之后使用数学公式即可完成。（我发现我自己的代码写的太菜了，题解的代码写的非常简洁QAQ，STL大法好）

    #include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
#include <cmath>
#include <queue>

using namespace std;

#define ll long long
#define db double
#define clr(a, b) memset(a, b, sizeof(a))
#define vi vector<int>

const ll mod = 1e9 + 7;
int n, t, len, i;
ll ans = 0;
int main()
{
    //freopen("out", "w", stdout);
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> t;
    while (t--)
    {
        string s;
        cin >> n >> s;
        ans = s.size();
        vector<char> _s(s.begin(), s.end());
        for (int i = 1; i <= n;i++)
        {
            int tmp = _s[i - 1] - '1';
            if(_s.size()<n)
            {
                vector <char> sub(_s.begin() + i, _s.end());
                for (int j = 1; j <= tmp;j++)
                    _s.insert(_s.end(), sub.begin(), sub.end());
              //  cout << _s << endl;
            }
            ans = (ans + tmp * (ans - i)) % mod;
        }
        cout << (ans+mod)%mod << endl;
    }
    return 0;
}

D. Beingawesomeism

  • 题意：
    给定一个行数为$r$ ，列数为$c$ 的由A和P组成的矩阵。可以进行多次操作。即选择一段连续的元素，一个方向，将该方向的所有元素都变成该段元素的复制。问最少进行多少次变化，可以使得该矩阵所有元素均为A。
  • 题解&代码：
    仔细分析一下，发现最多只需要四次操作即可。分类讨论即可。
int n;
int r, c;
int solve(int r,int c,vector<string>grid)
{
    vector<int> col(c), row(r);
    int tot = 0;
    for (int i = 0; i < r;i++)
            for (int j = 0; j < c;j++)
                if(grid[i][j]=='A')
                    row[i]++, col[j]++, tot++;
    if(tot==r*c)
        return 0;
    if(tot==0)
        return -1;
    if(row[0]==c || row[r-1]==c || col[0]==r || col[c-1]==r)
        return 1;
    if(*max_element(row.begin(),row.end())==c || 
        *max_element(col.begin(),col.end())==r ||
        grid.front().front()=='A' ||
        grid.back().front()=='A' ||
        grid.front().back()=='A' ||
        grid.back().back()=='A' 
        )
        return 2;
    if(row[0]>0 || row[r-1]>0 || col[0]>0 || col[c-1]>0)
        return 3;
    return 4;
}
int main()
{
    //freopen("out", "w", stdout);
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> n;
    while (n--)
    {
        cin >> r >> c;
        vector<string> grid(r);
        for (int i = 0; i < r;i++)
            cin >> grid[i];
        int ans = solve(r, c, grid);
        if(ans<0)
            cout << "MORTAL" << "\n";
        else
            cout << ans << "\n";
    }
    return 0;
}
