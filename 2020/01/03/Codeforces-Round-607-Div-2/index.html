struct node
{
    char x;
    int pos;

{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> n;
    while (n--)
    {
        cin >> s1 >> s2;
        int len = s1.length();
        for (int i = 0; i < len; i++)
            vec[i].x = vec[i].pos = 0;
        vec[len - 1].x = s1[len - 1];
        vec[len - 1].pos = len - 1;
        for (int i = len - 2; i >= 0; i--)
        {
            if (vec[i + 1].x > s1[i])
            {
                vec[i].x = s1[i];
                vec[i].pos = i;
            }
            else
            {
                vec[i].x = vec[i + 1].x;
                vec[i].pos = vec[i + 1].pos;
            }
        }
        for (int i = 0; i < len - 1; i++)
        {
            if (s1[i] > vec[i + 1].x)
            {
                int len = vec[i + 1].pos;
                swap(s1[i], s1[len]);
                break;
            }
        }
        if (s1 < s2)
            cout << s1 << endl;
        else
            cout << "---" << endl;
    }
    return 0;
}
