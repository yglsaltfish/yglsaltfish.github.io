A. Allocation

题意： 给定N，M，和N个数。求最多数量的数字，且这些数字之和不超过M。

题解： 排序，贪心。

#define ll long long
#define db double
#define clr(a, b) memset(a, b, sizeof(a))

const int maxn = 5e5 + 10;
const int inf = 0x7ffffff;

void solve(int T)
{
    int N, x, B;
    cin >> N >> B;
    vector<int> vec;
    for (int i = 1; i <= N;i++)
        cin >> x, vec.push_back(x);
    sort(vec.begin(), vec.end());
    int ans = 0;
    for (int i = 0; i < N;i++)
    {
        if(B-vec[i]>=0)
            ans++, B -= vec[i];
    }
    cout << "Case #" << T << ": " << ans << "\n";
}

int main()
{
    //freopen("out","w",stdout);
    ios::sync_with_stdio(false);
    cin.tie(0);
    int T = 1;
    //cin >> T;
    for (int i = 1; i <= T;i++)
        solve(i);

}

B. Plates

题意: n个栈，每个栈中有k个个元素，让你取p个元素，使得和最大

题解： 维护n个栈前缀和，记忆化搜索

#define ll long long
#define db double
#define clr(a, b) memset(a, b, sizeof(a))

const int maxn = 1600;

int ans = 0;
int N, P, K;
int prefix[55][maxn], record[55][maxn];
int mem[55][maxn];    //mem[x][y] 表示第x个栈之前已经选择了y个数
int dfs(int x ,int y)
{
    if(y < 0 || y > P || x < 1)
        return 0;
    if(mem[x][y]!=-1)
        return mem[x][y];
    for (int i = 0; i <= min(y, K); i++)
    {
        mem[x][y] = max(dfs(x - 1, y - i) + prefix[x][i], mem[x][y]);
    }
    return mem[x][y];

}

void solve(int T)
{
    cin >> N >> K >> P;
    clr(prefix, 0), clr(mem, -1);
    ans = 0;
    for (int i = 1; i <= N;i++)
    {
        for (int j = 1; j <= K;j++)
        {
            cin >> record[i][j];
            prefix[i][j] = prefix[i][j - 1] + record[i][j];
        }
    }
    cout << "Case #" << T << ": " << dfs(N, P) << "\n";
}

int main()
{
    //freopen("out","w",stdout);
    //freopen("in", "r", stdin);
    ios::sync_with_stdio(false);
    cin.tie(0);
    int T = 1;
   // cin >> T;
    for (int i = 1; i <= T;i++)
        solve(i);

}

C. Workout

题意: 给定N，K，N个数，N个数字中插入K个数，保持严格递增，求两两数字之间的差值最大的最小值。

题解：二分插值答案

#define ll long long
#define db double
#define clr(a, b) memset(a, b, sizeof(a))

const int maxn = 2e5 + 10;
const int inf = 0x7ffffff;

int M[maxn], N, K;
vector<int> vec;
int judge(int d)
{
    int cnt = 0;
    for (int i = 0; i < (int)vec.size();i++)
    {
        if(vec[i]%d==0)
            cnt = cnt + vec[i] / d - 1;
        else
            cnt = cnt + vec[i] / d;
    }
    if(cnt>K)
        return 0;
    return 1;
}

void solve(int T)
{
    int  dmax = -1;
    vec.clear();
    cin >> N >> K;
    for (int i = 1; i <= N; i++)
        cin >> M[i];
    for (int i = 2; i <= N; i++)
        vec.push_back(M[i] - M[i - 1]), dmax = max(dmax, M[i] - M[i - 1]);
    int l = 1, r = dmax, ans = 0, mid = 0;
    while (l<=r)
    {
        mid = (l + r) / 2;
        if(judge(mid))
            ans = mid, r = mid - 1;
        else
            l = mid + 1;
    }
    cout << "Case #" << T << ": " << ans << "\n";
}

int main()
{
    //freopen("out","w",stdout);
    ios::sync_with_stdio(false);
    cin.tie(0);
    int T = 1;
    cin >> T;
    for (int i = 1; i <= T; i++)
        solve(i);
}

D. Bundling

题意：给定N，K，N个字符串，N个字符串可以分成$ N /K$ 组，然后定一个值$ans=$ 所有租的前缀相同的字母的和，求这个值的最大值是多少。

题解：trie 计数.(google 编译器无法通过)

#define ll long long
#define db double
#define clr(a, b) memset(a, b, sizeof(a))

const int maxn = 1e5 + 10;
const int inf = 0x7ffffff;

struct Trie
{
    int next[maxn][26], ed[maxn];
    int par[maxn], dep[maxn];
    int L, root;
    int ans;
    int N, K;
    int newnode()
    {
        for (int i = 0; i < 26; i++)
            next[L][i] = -1;
        ed[L] = 0;
        return L++;
    }
    void init()
    {
        ans = L = 0;
        root = newnode();
    }
    void insert(char s[])
    {
        int now = root;
        for (int i = 0, sz = strlen(s); i < sz; i++)
        {
            if (next[now][s[i] - 'A'] == -1)
            {
                next[now][s[i] - 'A'] = newnode();
                par[next[now][s[i] - 'A']] = now;
                dep[next[now][s[i] - 'A']] = dep[now] + 1;
            }
            now = next[now][s[i] - 'A'];
        }
        ed[now]++;
    }
    int work()
    {
        int ans = 0;
        for (int i = L; i; i--)
        {
            ans += ed[i] / K * dep[i];
            ed[par[i]] += ed[i] % K;
        }
        return ans;
    }
};

Trie tree;
char s[maxn][maxn];

void solve(int T)
{

    tree.init();
    cin >> tree.N >> tree.K;
    for (int i = 0; i < tree.N; i++)
        cin >> s[i], tree.insert(s[i]);
    cout << "Case #" << T << ": " << tree.work() << "\n";
}

int main()
{
    //  freopen("in","r",stdin);
    ios::sync_with_stdio(false);
    cin.tie(0);
    int T = 1;
    cin >> T;
    for (int i = 1; i <= T; i++)
        solve(i);
}
