#define ll long long
#define db double
#define clr(a, b) memset(a, b, sizeof(a))

const int maxn = 2e5 + 10;
const int inf = 0x7ffffff;

int M[maxn], N, K;
vector<int> vec;
int judge(int d)
{
    int cnt = 0;
    for (int i = 0; i < (int)vec.size();i++)
    {
        if(vec[i]%d==0)
            cnt = cnt + vec[i] / d - 1;
        else
            cnt = cnt + vec[i] / d;
    }
    if(cnt>K)
        return 0;
    return 1;
}

void solve(int T)
{
    int  dmax = -1;
    vec.clear();
    cin >> N >> K;
    for (int i = 1; i <= N; i++)
        cin >> M[i];
    for (int i = 2; i <= N; i++)
        vec.push_back(M[i] - M[i - 1]), dmax = max(dmax, M[i] - M[i - 1]);
    int l = 1, r = dmax, ans = 0, mid = 0;
    while (l<=r)
    {
        mid = (l + r) / 2;
        if(judge(mid))
            ans = mid, r = mid - 1;
        else
            l = mid + 1;
    }
    cout << "Case #" << T << ": " << ans << "\n";
}

    int T = 1;
    cin >> T;
    for (int i = 1; i <= T; i++)
        solve(i);

#define ll long long
#define db double
#define clr(a, b) memset(a, b, sizeof(a))

const int maxn = 1e5 + 10;
const int inf = 0x7ffffff;

struct Trie
{
    int next[maxn][26], ed[maxn];
    int par[maxn], dep[maxn];
    int L, root;
    int ans;
    int N, K;
    int newnode()
    {
        for (int i = 0; i < 26; i++)
            next[L][i] = -1;
        ed[L] = 0;
        return L++;
    }
    void init()
    {
        ans = L = 0;
        root = newnode();
    }
    void insert(char s[])
    {
        int now = root;
        for (int i = 0, sz = strlen(s); i < sz; i++)
        {
            if (next[now][s[i] - 'A'] == -1)
            {
                next[now][s[i] - 'A'] = newnode();
                par[next[now][s[i] - 'A']] = now;
                dep[next[now][s[i] - 'A']] = dep[now] + 1;
            }
            now = next[now][s[i] - 'A'];
        }
        ed[now]++;
    }
    int work()
    {
        int ans = 0;
        for (int i = L; i; i--)
        {
            ans += ed[i] / K * dep[i];
            ed[par[i]] += ed[i] % K;
        }
        return ans;
    }
};

Trie tree;
char s[maxn][maxn];

void solve(int T)
{

    tree.init();
    cin >> tree.N >> tree.K;
    for (int i = 0; i < tree.N; i++)
        cin >> s[i], tree.insert(s[i]);
    cout << "Case #" << T << ": " << tree.work() << "\n";
}

int main()
{
    //  freopen("in","r",stdin);
    ios::sync_with_stdio(false);
    cin.tie(0);
    int T = 1;
    cin >> T;
    for (int i = 1; i <= T; i++)
        solve(i);
}
