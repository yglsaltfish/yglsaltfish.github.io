P1434 滑雪

  • 题意： 给定一个矩阵，求最长的递减数组的长度。

  • 题解: 这题目算是经典的模型了，如果使用搜索那么会有大量的对于重复问题的计算，那么就可以使用记忆化搜索来简化运算。

int n, m;
int dp[101][101] = {0};
int mp[101][101] = {0};
int dirx[] = {0, 1, 0, -1};
int diry[] = {1, 0, -1, 0};

int dfs(int ux,int uy)
{
    if (dp[ux][uy] != 1)
        return dp[ux][uy];
    for (int i = 0; i < 4;i++)
    {
        int tx = ux + dirx[i], ty = uy + diry[i];
        if (tx < 0 || tx > n || ty < 0 || ty > m)
            continue;
        if (mp[ux][uy] > mp[tx][ty])
        {
            dp[ux][uy] = max(dp[ux][uy], 1 + dfs(tx, ty));
        }
    }
    return dp[ux][uy];
}

void solve()
{
    cin >> n >> m;
    for (int i = 0; i < n;i++)
        for (int j = 0; j < m;j++)
            cin >> mp[i][j], dp[i][j] = 1;
    int res = 0;
    for (int i = 0; i < n;i++)
    {
        for (int j = 0; j < m; j++)
        {
            res = max(res, dfs(i, j));
        }
    }
    cout << res << "\n";
}

P2196 挖地雷

  • 题意： 给一个图，和图上每个点的权值，求最大的连续点的权值和，并且输出这些点。

  • 题解: 在动态规划的时候记录一下路径即可。

int mp[21][21] = {0};
int pre[21][21] = {0};
int val[21] = {0};
int dp[21] = {0};
int nxt[21] = {0};
int n;

void solve()
{
    cin >> n;
    memset(nxt, -1, sizeof(nxt));
    for (int i = 0; i < n;i++)
        cin >> val[i], dp[i] = val[i];
    for (int i = 0; i <= n;i++)
    {
        for (int j = i + 1; j < n; j++)
        {
            int ux;
            cin >> ux;
            if(ux)
                mp[i][j] = mp[j][i] = 1;
        }
    }
    for (int i = n - 1; i >= 0; i--)
    {
        for (int j = i + 1; j < n;j++)
        {
            if(mp[i][j])
            {
                if(dp[j]+val[i]>dp[i])
                {
                    dp[i] = max(dp[i], dp[j] + val[i]);
                    nxt[i] = j;
                }
                
            }
                
        }
    }
    int pos = 0, ans = 0;
    for (int i = 0; i < n;i++)
    {
        if (ans < dp[i])
            ans = dp[i], pos = i;
    }
    while (pos != -1)
    {
        cout << pos + 1 << " ";
        pos = nxt[pos];
    }
    cout << "\n"
         << ans;
}

P4017 最大食物链计数

  • 题意： 给一个图，求以入度为0的点为源点，出度为0的点为终点的路径的条数。

  • 题解: 记忆化搜索搜一下。

int n, m;
vector<vector<int>> mp;
vector<int> dp;
int vis[5005] = {0};
const int mod = 80112002;

int dfs(int u)
{
    if (dp[u])
        return dp[u];
    if(mp[u].size()==0)
        return 1;
    for(auto &it : mp[u])
    {
        dp[u] += dfs(it);
        dp[u] %= mod;
    }   
    return dp[u] % mod;
}

void solve()
{
    cin >> n >> m;
    mp.resize(n + 1);
    for (int i = 0; i < m;i++)
    {
        int ux, uy;
        cin >> ux >> uy;
        vis[ux]++;
        mp[uy].push_back(ux);
    }
    dp.resize(n + 1, 0);
    vector<int> source;
    int u = -1;
    for (int i = 1; i <= n;i++)
    {
        if (vis[i] == 0)
            source.push_back(i);
    }
    int res = 0;
    for(auto &it:source)
    {
        res += dfs(it);
        res %= mod;
    }
    cout << res << "\n";
}

P1280 尼克的任务

  • 题意： 尼克的一个工作日为 n 分钟，从第 1 分钟开始到第 n 分钟结束。当尼克到达单位后他就开始干活，公司一共有 k 个任务需要完成。如果在同一时刻有多个任务需要完成，尼克可以任选其中的一个来做，而其余的则由他的同事完成，反之如果只有一个任务，则该任务必需由尼克去完成，假如某些任务开始时刻尼克正在工作，则这些任务也由尼克的同事完成。问尼克的最大空闲时间为多少。

  • 题解: 规定\(dp[i]\)为从第\(i\)分钟开始，最大的空闲时间。那么就可以有转移方程

\[ dp(i)=\left\{\begin{aligned} &dp[i+1]+1,当前时刻没有任务开始\\ &max(dp[i],dp[i+task[i][j]]),当前时刻有任务开始，且任务的长度为task[i][j]\\ \end{aligned} \right. \]

int n, k;
vector<int> dp;
vector<int> task[10001];

void solve()
{
    cin >> n >> k;
    dp.resize(n + 3, 0);
    for (int i = 0; i < k;i++)
    {
        int ux, uy;
        cin >> ux >> uy;
        task[ux].push_back(uy);
    }
    int pos = 0;
    for (int i = n; i > 0; i--)
    {
        if(task[i].size())
        {
            for (int j = 0; j < task[i].size();j++)
                dp[i] = max(dp[i], dp[i + task[i][j]]);
        }
        else
            dp[i] = dp[i + 1] + 1;
    }
    cout << dp[1] << "\n";
}

P2758 编辑距离

  • 题意： 给定两个字符串，要用最少的字符操作次数，将字符串A转换为字符串B。有三种操作方式，分别是删除，插入和替换。

  • 题解: 定义\(dp[i][j]\)表示将A字符串的\(0 - i\) 的字符串变换到和B字符串的\(0 - j\)相同的最少字符操作次数。

\[ dp[i][j]=\left\{\begin{aligned} &dp[i-1][j-1],s[i-1]==t[j-1]\\ &min(dp[i-1][j-1],dp[i][j-1]),dp[i-1][j])+1,s[i-1]\not ={t[j-1]}\\ \end{aligned} \right. \]

string s, t;
vector<vector<int>> dp(2003, vector<int>(2003, 0));

void solve()
{
    cin >> s >> t;
    int ls = s.size(), lt = t.size();
    s += " ", t += " ";
    for (int i = 1; i <= ls;i++)
        dp[i][0] = i;
    for (int i = 1; i <= lt; i++)
        dp[0][i] = i;
    for (int i = 1; i <= ls; i++)
    {
        for (int j = 1; j <= lt; j++)
        {
            int k = (s[i - 1] == t[j - 1] ? 0 : 1);
            dp[i][j] = min(dp[i - 1][j - 1] + k, min(dp[i - 1][j] + 1, dp[i][j - 1] + 1));
        }
    }
    cout << dp[ls][lt] << "\n";
}

P1040 加分二叉树

  • 题意： 设定一个节点数为n的二叉树，每个节点都有权值，并且二叉树的中序遍历为\(1,2,3,...,n\)，tree 及它的每个子树都有一个加分，任一棵子树 加分计算方法为: 左子树的加分 * 右子树的加分 + 本身的分数。若某个子树为空，规定其加分为 1，叶子的加分就是叶节点本身的分数。不考虑它的空子树。

  • 题解: 在中序遍历这个序列上，某个点左边的序列一定是这个点的左子树，右边的序列，一定在这个点的右子树。 定义\(dp[i][j]\)表示从节点i到节点j的最大分数，枚举i到J的根节点K，动态更新即可。 转移方程：

    \(dp[i][j] = max(dp[i][k-1]*dp[k+1][j]+value[k],dp[i][j])\)

int n;
vector<vector<long long>> dp(40, vector<long long>(40, 0));
vector<vector<long long>> rt(40, vector<long long>(40, 0));
vector<long long> value(40, 0);

void dfs(int st,int ed)
{
    if(st==ed)
    {
        cout << st << " ";
        return;
    }
    if(rt[st][ed]==0)
        return;
    cout << rt[st][ed] << " ";
    dfs(st, rt[st][ed] - 1);
    dfs(rt[st][ed] + 1, ed);
}

void solve()
{
    cin >> n;
    for (int i = 1; i <= n;i++)
    {
        cin >> value[i];
        dp[i][i] = value[i];
        dp[i][i - 1] = 1;
    }
    for (int len = 1; len < n; len++)
    {
        for (int i = 1; i + len <= n; i++)
        {
            int j = i + len;
            for (int k = i; k <= j;k++)
            {
                if (dp[i][j] < dp[i][k - 1] * dp[k + 1][j] + dp[k][k])
                {
                    dp[i][j] = max(dp[i][j], dp[i][k - 1] * dp[k + 1][j] + dp[k][k]);
                    rt[i][j] = k;
                }
            }
        }
    }
    cout << dp[1][n] << "\n";
    dfs(1, n);
}

P4933 大师

  • 题意： 给定一串序列，求子序列中为等差数列的个数。

  • 题解: 动态规划，\(dp[i][j]\) 表示到以第i个元素为结尾，公差为j的序列的个数。转移方程:

    \(dp[i][value[i]-value[j]] += dp[j][value[i]-value[j]]+1\)

vector<vector<int>> dp(1005, vector<int>(40005, 0));
int n;
vector<ll> value;

void solve()
{
    cin >> n;
    value.resize(n + 2, 0);
    ll res = 0;
    for (int i = 1; i <= n; i++)
        cin >> value[i];
    int p = 20002;
    for (int i = 1; i <= n;i++)
    {
        for (int j = i - 1; j >= 1; j--)
        {
            dp[i][p + value[i] - value[j]] += dp[j][p + value[i] - value[j]] + 1;
            dp[i][p + value[i] - value[j]] %= mod;
            res += dp[j][p + value[i] - value[j]] + 1;
            res %= mod;
        }
    }
    cout << res + n << "\n";
}

P1077 摆花

  • 题意： 有n个物品，每个物品有\(a[i]\)个，按顺序摆放m个，求摆放的种类的数量。

  • 题解: 动态规划，\(dp[i][j]\) 表示到第i个元素，放了j个物品的摆放种类，枚举j即可。

    \(dp[i][j+k] += dp[i-1][k]\)

int n, m;
vector<int> vec;
vector<vector<int>> dp(105, vector<int>(105, 0));
void solve()
{
    cin >> n >> m;
    vec.resize(n + 1);
    for (int i = 1; i <= n;i++)
        cin >> vec[i];
    for (int i = 0; i <= n;i++)
        dp[i][0] = 1;
    for (int i = 1; i <= n;i++)
    {
        for (int j = 0; j <= vec[i]; j++)
        {
            for (int k = 0; k + j <= m;k++)
            {
                if (j == 0 && k == 0)
                    continue;
                dp[i][j + k] += dp[i - 1][k];
                dp[i][j + k] %= mod;
            }
        }
    }
    cout << dp[n][m] << "\n";
}

LeetCode 解出数学表达式的学生分数

  • 题意： 给定一个由个位数数字、'\(+\)'、'\(*\)'组成的表达式，给定一个数组\(answers[\quad]\)，如果一答案等于表达式的正确结果，那么将得到 5 分。否则，如果答案由一处或多处错误的运算顺序计算得到那么将得到 2 分。求最后分数。

  • 题解: 一开始以为暴力能解，发现行不通，考虑使用区间DP。 考虑先计算出正确值，然后再计算错误运算顺序得到的值。对于一个较大的区间，枚举最后一次的计算符号，记录即可。 转移方程为：

    \(dp[i][i+step] insert(dp[i+t] \circ dp[i+t+2][i+step])\) 其中\(\circ\) 表示 \(+\) 或者 \(*\) ，step为枚举的区间大小，t为枚举的最后一次运算顺序。

class Solution
{
public:
    int cal(string &s)
    {
        
        char sign = '+';
        char c;
        int num = 0, sum = 0;
        stack<int> st;
        for (int i = 0; i < s.size(); i++)
        {
            if (s[i] >= '0' && s[i] <= '9')
                num = num * 10 + s[i] - '0';
            if (s[i] == '+' || s[i] == '*' || i == (int)s.size() - 1)
            {
                if (sign == '*')
                {
                    int tp = st.top();
                    st.pop();
                    st.push(tp * num);
                }
                else
                    st.push(num);
                num = 0;
                sign = s[i];
            }
        }
        while (st.size())
        {
            sum += st.top();
            st.pop();
        }
        return sum;
    }
    int scoreOfStudents(string s, vector<int> &answers)
    {
        vector<int> mp(1030, 0);
        for (auto &it : answers)
            mp[it]++;
        int len = s.size();
        int res = 3 * mp[cal(s)];
        vector<vector<unordered_set<int>>> dp(len + 2, vector<unordered_set<int>>(len + 2));
        for (int i = 0; i < len;i++)
            dp[i][i].insert(s[i] - '0');
        for (int step = 2; step < len; step += 2)
        {
            for (int i = 0; i + step < len; i += 2)
            {
                for (int t = 0; t < step; t += 2)
                {
                    for (auto &x : dp[i][i + t])
                    {
                        for (auto &y : dp[i + t + 2][i + step])
                        {
                            if(s[i+t+1]=='+')
                            {
                                if (x + y <= 1000)
                                    dp[i][i + step].insert(x + y);
                            }
                            else
                            {
                                if (x * y <= 1000)
                                    dp[i][i + step].insert(x * y);
                            }
                        }
                    }
                }
            }
        }
        for (auto &x : dp[0][len - 1])
            res += 2 * mp[x];
        return res;
    }
};

P1233 木棍加工

  • 题意： 求最少最长不上升的子序列的数目。

  • 题解: 根据Dilworth定理，最少最长不升子序列的数目=最长上升子序列的长度。

int n;
vector<int> dp(5005, 0);
vector<pair<int, int>> vec;

void solve()
{
    cin >> n;
    vec.resize(n);
    for (int i = 0; i < n; i++)
        cin >> vec[i].first >> vec[i].second;
    sort(vec.rbegin(), vec.rend());
    int ans = 0;
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < i; j++)
        {
            if (vec[i].second > vec[j].second)
                dp[i] = max(dp[i], dp[j] + 1);
        }
        ans = max(dp[i], ans);
    }
    cout << ans + 1 << "\n";
}

P1220 关路灯

  • 题意： 求最少的耗电

  • 题解: 考虑区间DP。 \(dp[i][j][0]\)表示区间i到j的灯都被关闭了，且站在第i个点； \(dp[i][j][1]\)表示区间i到j的灯都被关闭了，且站在第j个点； 则有状态转移方程：

    \(dp[i][j][0] = min(dp[i + 1][j][0] + cal(i, i + 1, i, j), dp[i + 1][j][1] + cal(i, j, i, j));\)

    \(dp[i][j][1] = min(dp[i][j - 1][1] + cal(j - 1, j, i - 1, j - 1), dp[i][j - 1][0] + cal(i, j, i - 1, j - 1));\)

vector<pair<int, int>> vec;
vector<int> prefix;
int cal(int u, int v, int l, int r)
{
    return (vec[v].first - vec[u].first) * (prefix[n] + prefix[l] - prefix[r]);
}

P3205 合唱队

  • 题意： 求初始队形的数量。

  • 题解: 考虑使用区间DP。 \(dp[i][j][0]\)表示区间\([i,j]\)且第i个元素从左边进入的方案数。\(dp[i][j][1]\)表示区间\([i,j]\)且第j个元素从右边进入的方案数。 则存在动态转移方程：

    \[ dp[i][j][0] +=\left\{\begin{aligned} &dp[i+1][j][0],num[i]<num[i+1]\\ &dp[i+1][j][1],num[i]<num[j]\ \end{aligned} \right. \]

    \[ dp[i][j][1] +=\left\{\begin{aligned} &dp[i][j-1][1],\quad num[j]>num[i]\\ &dp[i][j-1][0],\quad num[j]>num[i] \\ \end{aligned} \right. \]

int dp[1005][1005][3] = {0};
int n;

void solve()
{
    cin >> n;
    vector<int> num(n + 1);
    for (int i = 1; i <= n; i++)
        cin >> num[i];
    for (int i = 1; i <= n; i++)
        dp[i][i][1] = 1;
    for (int len = 1; len <= n; len++)
    {
        for (int i = 1; i + len <= n; i++)
        {
            int j = i + len;
            if (num[i] < num[i + 1])
                dp[i][j][0] += dp[i + 1][j][0];
            if (num[i] < num[j])
                dp[i][j][0] += dp[i + 1][j][1];
            if (num[j] > num[j - 1])
                dp[i][j][1] += dp[i][j - 1][1];
            if (num[j] > num[i])
                dp[i][j][1] += dp[i][j - 1][0];
            dp[i][j][0] %= mod;
            dp[i][j][1] %= mod;
        }
    }
    cout << (dp[1][n][0] + dp[1][n][1]) % mod << "\n";
}

P1880 石子合并

  • 题意： 求合并的最大值和最小值

  • 题解: 很经典的问题，考虑使用区间DP，注意是环形石子； \(dp[i][j][0]\)表示区间\([i,j]\)合并的最小的得分。\(dp[i][j][1]\)表示区间\([i,j]\)合并的最大得分。 则存在动态转移方程：

    \(dp[i][j][0]=min(dp[i][k][0]+dp[k+1][j][0]+prefix[j]-prefix[i-1],dp[i][j][0])\)

    \(dp[i][j][1]=max(dp[i][k][1]+dp[k+1][j][1]+prefix[j]-prefix[i-1],dp[i][j][1])\)

int dp[105 << 1][105 << 1][3];
int n;
int num[105 << 1] = {0};
int prefix[105 << 1] = {0};

void solve()
{
    cin >> n;
    for (int i = 1; i <= n; i++)
    {
        cin >> num[i];
        num[i + n] = num[i];
    }
    for (int i = 1; i <= (n << 1); i++)
    {
        prefix[i] = prefix[i - 1] + num[i];
    }
    for (int i = 0; i <= 2 * n; i++)
    {
        for (int j = 0; j <= 2 * n; j++)
            dp[i][j][0] = inf;
        dp[i][i][0] = dp[i][i][1] = 0;
    }
    for (int len = 1; len < n; len++)
    {
        for (int i = 1; i + len <= 2 * n; i++)
        {
            int j = i + len, tmp = inf;
            for (int k = i; k < j; k++)
            {
                tmp = min(dp[i][k][0] + dp[k + 1][j][0] + prefix[j] - prefix[i - 1], tmp);
                dp[i][j][1] = max(dp[i][k][1] + dp[k + 1][j][1] + prefix[j] - prefix[i - 1], dp[i][j][1]);
            }
            dp[i][j][0] = min(dp[i][j][0], tmp);
        }
    }
    int maxx = 0, minn = inf;
    for (int i = 1; i <= n; i++)
    {
        maxx = max(maxx, dp[i][i + n - 1][1]);
        minn = min(minn, dp[i][i + n - 1][0]);
    }

    cout << minn << "\n";
    cout << maxx << "\n";
}

P1063 能量项链

  • 题意： 求合并的权值最大的值

  • 题解： 一眼贪心，但是细想不行。有点像石子合并，考虑区间DP;但是这体面没给数据范围，数组只能往大了开。

    \(dp[i][j]\)表示区间i和j所能得到的最大值，那么只需要考虑枚举\([i,j]\)之间的值即可，转移方程为: \(dp[i][j]=max(dp[i][j],dp[i][k]+dp[k][j]+num[i]*num[k]*num[j])\)

int n;

{

        {

            {

            }
        }

}

P3146 248G

  • 题意： 给定一个数组，期间相等的数字可以两两合并，问最大的合并出的数字是多少。

  • 题解： 这题可以一眼看出是区间DP。\(dp[i][j]\)表示区间\([i,j]\)能合并出的最大的值。转移方程：

    枚举\([i,j]\)之间的\(k\)，当\(dp[i][k]==dp[k+1][j]\) 时 \(dp[i][j]=max(dp[i][k]+1,dp[i][j])\)

int n;
int dp[280][280];
int num[280];

void solve()
{
    cin >> n;
    for (int i = 1; i <= n; i++)
    {
        cin >> num[i];
        dp[i][i] = num[i];
    }

    for (int len = 1; len < n; len++)
    {
        for (int i = 1; i + len <= n; i++)
        {
            int j = i + len;
            for (int k = i; k < j; k++)
            {
                if (dp[i][k] == dp[k + 1][j])
                    dp[i][j] = max(dp[i][j], dp[i][k] + 1), res = max(dp[i][j], res);
            }
        }
    }

}

P4170 涂色

  • 题意： 求最少的染色次数。

  • 题解： \(dp[i][j]\)表示区间\([i,j]\)为一个颜色的最少涂色次数。状态转移方程为：

    \[ dp[i][j]=\left\{\begin{aligned} &min(dp[i+1][j],dp[i][j-1]),if(s[i]==s[j])\\ &min(dp[i][j],dp[i][k]+dp[k+1][j]),k \in [i,j) \end{aligned} \right. \]

const int maxn = 5e5 + 10;
const int inf = 0x7fffff;

int dp[105][105];

void solve()
{
    string s;
    cin >> s;
    int n = s.size();
    s = "0" + s;
    for (int i = 0; i <= n; i++)
    {
        for (int j = 0; j <= n;j++)
            dp[i][j] = inf;
        dp[i][i] = 1;
    } 
    for (int len = 1; len <= n; len++)
    {
        for (int i = 1; i + len <= n; i++)
        {
            int j = i + len;
            if (s[i] == s[j])
                dp[i][j] = min(dp[i + 1][j], dp[i][j - 1]);
            else
            {
                for (int k = i; k < j;k++)
                {
                    dp[i][j] = min(dp[i][j], dp[i][k] + dp[k + 1][j]);
                }
            }
        }
    }
    cout << dp[1][n] << "\n";
}

CF 607B

  • 题意： 定义每次操作能消除一个回文子序列，问消除所有数组的最小操作数是多少。

  • 题解： \(dp[i][j]\)表示消除区间\([i,j]\)的最小操作数是多少，则存在状态转移方程：

    \[ \left\{\begin{aligned} &dp[i][i]=1\\ &dp[i][i+1]=1+(num[i]\not ={num[i+1]})\\ &dp[i][j]=min(dp[i][k]+dp[k+1][j],dp[i][j]),k \in [i,j) \end{aligned} \right. \]

int dp[maxn][maxn];
int num[maxn];
int n;

void solve()
{
    cin >> n;
    for (int i = 1; i <= n; i++)
        cin >> num[i];
    for (int i = 0; i <= n; i++)
    {
        for (int j = 0; j <= n; j++)
            dp[i][j] = inf;
    }
    for (int i = 1; i <= n; i++)
    {
        dp[i][i] = 1;
        dp[i][i + 1] = 1 + (num[i] != num[i + 1]);
    }
    dp[n][n + 1] = inf;
    for (int len = 2; len <= n; len++)
    {
        for (int i = 1; i + len <= n; i++)
        {
            int j = i + len;
            if (num[i] == num[j])
                dp[i][j] = dp[i + 1][j - 1];
            for (int k = i; k < j; k++)
            {
                dp[i][j] = min(dp[i][k] + dp[k + 1][j], dp[i][j]);
            }
        }
    }
    cout << dp[1][n] << "\n";
}

P1352 没有上司的舞会

  • 题意： 给一棵树和树上n个节点的权值，父节点和子节点不能同时取，问最大权值和是多少。

  • 题解： \(dp[i][0]\)表示不取第\(i\)个节点的最大权值和，\(dp[i][1]\)表示取第\(i\)个节点的最大权值和则有状态转移方程:

    \[ \left\{\begin{aligned} &dp[u][0]+=max(dp[v][0],dp[v][1]) \\ &dp[u][1]+=dp[v][0] \end{aligned} \right. \]

int n;
int dp[maxn][3] = {0};
int num[maxn];
vector<vector<int>> rec;
int vis[maxn] = {0};
void dfs(int rt)
{
    dp[rt][0] = 0;
    dp[rt][1] = num[rt];
    for (auto &it : rec[rt])
    {
        dfs(it);
        dp[rt][0] += max(dp[it][1], dp[it][0]);
        dp[rt][1] += dp[it][0];
    }
}
void solve()
{
    cin >> n;
    for (int i = 1; i <= n;i++)
        cin >> num[i];
    rec.resize(n + 1);
    for (int i = 1; i < n;i++)
    {
        int x, y;
        cin >> x >> y;
        rec[y].push_back(x);
        vis[x] = 1;
    }
    int rt = 0;
    for (int i = 1; i <= n;i++)
    {
        if(vis[i]==0)
        {
            rt = i;
            break;
        }
    }

    dfs(rt);
    cout << max(dp[rt][0], dp[rt][1]) << "\n";
}

P2015 二叉苹果树

  • 题意： 给一棵树和树上\(n-1\)个边的权值，问如何取边才能做到最大化权值和。

  • 题解： \(dp[i][j]\)第\(i\)个节点下的子树取 \(k\) 个边所能取到的最大权值和。则有状态转移方程:

\(dp[u][k]=max(dp[u][k],dp[v][j]+dp[u][k-j-1]+mp[u][v]),k \in [1,q],j \in [0,k-1]\)

int n, q;
int mp[105][105];
int dp[105][105];
int vis[105] = {0};
vector<vector<int>> vec;

void dfs(int rt)
{
    vis[rt] = 1;
    for (int i = 0; i < vec[rt].size(); i++)
    {
        if (vis[vec[rt][i]])
            continue;
        dfs(vec[rt][i]);
        for (int k = q; k >= 1; k--)
        {
            for (int j = k - 1; j >= 0; j--)
            {
                dp[rt][k] = max(dp[rt][k], dp[vec[rt][i]][j] + dp[rt][k - j - 1] + mp[rt][vec[rt][i]]);
            }
        }
    }
}

void solve()
{
    cin >> n >> q;
    int x, y, z;
    vec.resize(n + 1);
    for (int i = 0; i < n-1; i++)
    {
        cin >> x >> y >> z;
        mp[x][y] = z, mp[y][x] = z;
        vec[x].push_back(y), vec[y].push_back(x);
    }
    dfs(1);
    cout << dp[1][q] << "\n";
}

P1613 跑路

  • 题意： 按题目描述花最短时间跑路。

  • 题解： 由于跑 \(2^k\) 花费最少，那么只要将每个跑 \(2^k\) 的路径预处理花费为1，最后跑一遍最短路即可。

int n, m;
int mp[60][60];
int dis[60][60];
int dp[60][60][70] = {0};

void solve() 
{
    int u, v;
    cin >> n >> m;
    for (int i = 0; i <= n + 1;i++)
        for (int j = 0; j <= n + 1;j++)
            dis[i][j] = inf;
    for (int i = 0; i < m;i++)
    {
        cin >> u >> v;
        dis[u][v] = 1;
        dp[u][v][0] = 1;
    }
    for (int o = 1; o <= 64; o++)
    {
        for (int k = 1; k <= n; k++)
        {
            for (int i = 1; i <= n; i++)
            {
                for (int j = 1; j <= n; j++)
                {
                    if (dp[i][k][o - 1] && dp[k][j][o - 1])
                    {
                        dp[i][j][o] = 1;
                        dis[i][j] = 1;
                    }
                }
            }
        }
    }
        
    for (int k = 1; k <= n;k++)
    {
        for (int i = 1; i <= n;i++)
        {
            for (int j = 1; j <= n;j++)
            {
                dis[i][j] = min(dis[i][j], dis[i][k] + dis[k][j]);
            }
        }
    }
    cout << dis[1][n] << "\n";
}

P2656 采蘑菇

  • 题意： 按题意采蘑菇。

  • 题解： 这道题目给我做麻了ORZ....最后会有一个精度问题。 总的来说还是比较简单的，就是tarjan缩点+拓扑排序(DAG上求最长路)

#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define db double
#define clr(a, b) memset(a, b, sizeof(a))

const int inf = 0x7fffffff;

const int maxn = 3e5 + 10;

int low[maxn], dfn[maxn], tot = 0, tot2 = 0;
int head[maxn], head2[maxn], vis[maxn];
int team[maxn], team_value[maxn], in[maxn];
int n, m, s;

struct node
{
    int from;
    int to, w, next;
    db reco;

} edge[maxn], edge2[maxn];

void addedge(int x, int y, int z, db recover)
{
    tot++;
    edge[tot].from = x;
    edge[tot].w = z, edge[tot].to = y, edge[tot].next = head[x];
    edge[tot].reco = recover;
    head[x] = tot;
}

void addedge2(int x, int y, int z)
{
    tot2++;
    edge2[tot2].from = x;
    edge2[tot2].w = z, edge2[tot2].to = y, edge2[tot2].next = head2[x];
    head2[x] = tot2;
}

stack<int> st;
int nd = 0;
int tot_team = 0;
int cal(int x, db y)
{
    int sum = 0;
    while (x)
    {
        sum += x;
        x = floor(1.0 * y * x);
    }
    return sum;
}
void tarjan(int x)
{
    dfn[x] = low[x] = ++nd;
    vis[x] = 1;
    st.push(x);
    for (int i = head[x]; i; i = edge[i].next)
    {
        int v = edge[i].to;
        if (!dfn[v])
        {
            tarjan(v);
            low[x] = min(low[x], low[v]);
        }
        else if (vis[v])
        {
            low[x] = min(low[x], dfn[v]);
        }
    }

    if (low[x] == dfn[x])
    {
        int cnt = 0;
        tot_team++;
        while (1)
        {
            int now = st.top();
            st.pop();
            vis[now] = 0;
            team[now] = tot_team;
            if (now == x)
                break;
        }
    }
}

int dp[maxn];

void topo(int s)
{
    queue<int> que;
    for (int i = 1; i <= tot_team; i++)
    {
        if (in[i] == 0)
            que.push(i);
        dp[i] = -inf;
    }
    dp[team[s]] = team_value[team[s]];
    while (que.size())
    {
        int tp = que.front();
        que.pop();
        for (int u = head2[tp]; u; u = edge2[u].next)
        {
            int v = edge2[u].to;
            dp[v] = max(dp[v], dp[tp] + edge2[u].w + team_value[v]);
            in[v]--;
            if (in[v] == 0)
                que.push(v);
        }
    }
    int ans = INT_MIN;
    for (int i = 1; i <= tot_team; i++)
    {
        ans = max(ans, dp[i]);
    }
        
    cout << ans << "\n";
}

void solve()
{
    cin >> n >> m;
    int u, v, ini;
    db re;
    for (int i = 1; i <= m; i++)
    {
        cin >> u >> v >> ini >> re;
        re += 1e-6;
        addedge(u, v, ini, re);
    }
    for (int i = 1; i <= n; i++)
    {
        if (!dfn[i])
            tarjan(i);
    }

    for (int i = 1; i <= m; i++)
    {
        if (team[edge[i].from] != team[edge[i].to])
        {
            addedge2(team[edge[i].from], team[edge[i].to], edge[i].w);
            in[team[edge[i].to]]++;
        }
        else
        {
            team_value[team[edge[i].from]] += cal(edge[i].w, edge[i].reco);
        }
    }

    cin >> s;
    topo(s);
}

P1122 最大子树和

  • 题意： 按题意求最大子树和。

  • 题解： 比较简单的一道树形DP，\(dp[i]\)表示第i个子树（包括根节点）的最大子树和。转移方程： \(dp[u] += (dp[v]>0 \, ? \, dp[v] \, : \, 0)\)，v是u的子节点

int n;
vector<int> mp[maxn];
int a[maxn];
int dp[maxn] = {0};

void dfs(int rt, int fa)
{
    dp[rt] = a[rt];
    for (int &it : mp[rt])
    {
        if (it == fa)
            continue;
        dfs(it, rt);
        if (dp[it] > 0)
            dp[rt] += dp[it];
    }
}

void solve()
{
    cin >> n;
    for (int i = 1; i <= n; i++)
    {
        cin >> a[i];
    }
    int u, v;
    for (int i = 1; i < n; i++)
    {
        cin >> u >> v;
        mp[u].push_back(v);
        mp[v].push_back(u);
    }
    dfs(1, 0);
    int ans = INT_MIN;
    for (int i = 1; i <= n;i++)
        ans = max(dp[i], ans);
    cout << ans << "\n";
}
