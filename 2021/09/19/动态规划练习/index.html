int dirx[] = {0, 1, 0, -1};
int diry[] = {1, 0, -1, 0};

int dfs(int ux,int uy)
{
    if (dp[ux][uy] != 1)
        return dp[ux][uy];
    for (int i = 0; i < 4;i++)
    {
        int tx = ux + dirx[i], ty = uy + diry[i];
        if (tx < 0 || tx > n || ty < 0 || ty > m)
            continue;
        if (mp[ux][uy] > mp[tx][ty])
        {
            dp[ux][uy] = max(dp[ux][uy], 1 + dfs(tx, ty));
        }
    }
    return dp[ux][uy];
}

void solve()
{
    cin >> n >> m;

        for (int j = 0; j < m;j++)

    int res = 0;
    for (int i = 0; i < n;i++)
    {
        for (int j = 0; j < m; j++)
        {
            res = max(res, dfs(i, j));
        }
    }

}

        return res;
    }
